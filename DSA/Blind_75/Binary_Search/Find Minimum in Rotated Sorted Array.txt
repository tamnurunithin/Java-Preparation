1.Finding Minimum means making left=right where left is minimum

class Main {
    public static void main(String[] args) {
        int[] nums={4,5,6,7,8,1,2,3};
        int n =nums.length;
        int left=0;
        int right=n-1;
        while(left<right){
            int mid = left+(right-left)/2;
            if(nums[mid]>nums[right]){
                left=mid+1;
            }else{
                right=mid;
            }
        }
        System.out.print(nums[left]);
    }
}
