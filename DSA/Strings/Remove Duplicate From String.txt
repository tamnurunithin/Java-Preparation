class Main {
    public static void main(String[] args) {

        String s = "NithiN";   // input
        int[] freq = new int[256];
        String result = "";

        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);

            if (freq[ch] == 0) {   // first time seeing this char
                result = result + ch;
                freq[ch] = 1;
            }
        }

        System.out.println(result);
    }
}
