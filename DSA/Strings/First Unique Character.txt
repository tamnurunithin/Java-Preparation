class Main {
    public static void main(String[] args) {
        String str="nithin";
        int n = str.length();
        int[] freq=new int[26];
        int foundindex=-1;
        for(int i=0;i<n;i++){
            char ch = str.charAt(i);
            freq[ch-'a']++;
        }
        for(int i=0;i<n;i++){
            char ch = str.charAt(i);
            if(freq[ch-'a']==1){
                foundindex=i;
                System.out.println(ch+" "+i);
                break;
            }
        }
        if(foundindex==-1){
            System.out.println("No Unique Element");
        }
    }
}
