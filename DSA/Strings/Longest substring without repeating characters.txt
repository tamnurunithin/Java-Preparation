class Main {
    public static void main(String[] args) {
        String s = "abcabcbb";
        int n = s.length();
        int[] freq = new int[256];
        int max=0;
        int left =0;
        for(int right=0;right<n;right++){
            char ch = s.charAt(right);
            freq[ch]++;
            while(freq[ch]>1){
                freq[s.charAt(left)]--;
                left++;
            }
            int len=right-left+1;
            if(len>max){
                max=len;
            }
        }
        System.out.println("MaxLength Of Substring : "+max);
    }
}
