class Main {
    public static void main(String[] args) {
        String s1="abc";
        String s2="def";
        String s3="adbecf";
        int n1=s1.length();
        int n2=s2.length();
        int n3=s3.length();
        int i=0; 
        int j=0;
        int k=0;
        if(n3!=n1+n2){
            System.out.println("valid Shuffle");
            return;
        }
        while(k<n3){
            if(i<n1 && s3.charAt(k)==s1.charAt(i)){
                i++;
            }
            else if(j<n2 && s3.charAt(k)==s2.charAt(j)){
                j++;
            }else{
            System.out.println("Not valid Shuffle");
            return;
            }
            k++;
        }
        if (i == n1 && j == n2){
            System.out.println("Valid shuffle");
        }else {
            System.out.println("Not a valid shuffle"); }
    }
    
}
