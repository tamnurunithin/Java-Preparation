class Solution {
    //Scan and calculate the Sum of Squares Of Digits
    private int DigitSumSquare(int n){
        int sum =0;
        while(n>0){
            int digit=n%10;
            sum+=digit*digit;
            n/=10;
        }
        return sum;

    }
    public boolean isHappy(int n) {
        //Create Hashmaps to store sum of the squres of value equals one and Boolean
        HashMap<Integer,Boolean> map=new HashMap<>();
        while(n!=1){
            if(map.containsKey(n)){
                return false;
            }
            map.put(n,true);
            n=DigitSumSquare(n);
        }
        return true;

        
    }
}
