class Main {
    public static void main(String[] args) {
        int[] arr={1,2,3,4,5};
        int k=2;
        int n=arr.length;
        k=k%n;
        int[] temp=new int[n];
        int index=0;
        for(int i=k;i<n;i++){
            temp[index++]=arr[i];
        }
        for(int i=0;i<k;i++){
            temp[index++]=arr[i];
        }
        for(int i=0;i<n;i++){
            arr[i]=temp[i];
        }
        for(int i=0;i<n;i++){
            System.out.println(arr[i]);
        }
    }
}
