âš¡Approch:
1.Traverse by using i and j and add elements into sum
2.If sum equals to target 
3.Find length j-i+1
4.If length greater than maxlength update maxlength

class Main {
    public static void main(String[] args) {
        int[] arr ={10,5,2,7,1,9};
        int k =15;
        int n = arr.length;
        int maxlen=0;
        for(int i=0;i<n;i++){
            int sum=0;
            for(int j =i;j<n;j++){
                sum=sum+arr[j];
                if(sum==k){
                    int len=j-i+1;
                    if(len>maxlen){
                        maxlen=len;
                    }
                }
            }
        }
        System.out.println(maxlen);
    }
}
