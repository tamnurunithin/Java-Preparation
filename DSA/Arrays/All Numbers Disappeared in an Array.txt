class Main {
    public static void main(String[] args) {

        int[] nums = {4,3,2,7,8,2,3,1};
        int n = nums.length;

        int[] freq = new int[n];  to n

        
        for (int i = 0; i < n; i++) {
            freq[nums[i] - 1]++; // 
        }
        for (int i = 0; i < n; i++) {
            if (freq[i] == 0) {
                System.out.print((i + 1) + " ");
            }
        }
    }
}
