class Main {
    public static void main(String[] args) {
        int[] arr = {1, 3, 4, 2, 2};  // N+1 size
        int n = arr.length - 1;

        int expectedSum = n * (n + 1) / 2;
        int actualSum = 0;

        for (int i = 0; i < arr.length; i++) {
            actualSum += arr[i];
        }

        int duplicate = actualSum - expectedSum;
        System.out.println("Duplicate element: " + duplicate);
    }
}
