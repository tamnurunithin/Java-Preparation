class Solution {
    public int lengthOfLastWord(String s) {
int length =0;
int i = s.length()-1;
//Remove Spaces
while(i>=0 && s.charAt(i)==' '){
i--;
}
//Count Last Word
while(i>=0 && s.charAt(i)!=' '){
length++;
i--
}
return length;
}
}
