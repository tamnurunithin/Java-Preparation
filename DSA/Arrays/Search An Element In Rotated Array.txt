1.Initialise low,high,target,index=-1
2.Run a while loop until low less than equals to high
3.Find the mid value and check it matches with target and find index
4.Check on left side:
	If left value less than right value then it is sorted
	If target>low and less than mid update high and low
5.Else:
	If target>mid and less than equals to high update low and high
6.Print the Index

class Main {
    public static void main(String[] args) {
        int [] arr ={4,5,6,7,0,1,2};
        int target = 0;
        int n = arr.length;
        int low=0;
        int high=n-1;
        int index=-1;
        
        while(low<=high){
            int mid = (low+high)/2;
            if(arr[mid]==target){
                index=mid;
                break;
            }
            if(arr[low]<=arr[mid]){
                if(target>=arr[low] && target<arr[mid]){
                    high = mid-1;
                }else{
                    low=mid+1;
                }
                }else{
                    if(target>arr[mid] && target<=arr[high]){
                        low=mid+1;   
                    }else{
                        high=mid-1; 
                    } 
                }
        }
        System.out.println(index);
    }
}
