1.Create No of rows

3.Intialise first and last element of each row as 1

class Main {
    public static void main(String[] args) {
        int rows=5;
        int[][] triangle = new int[rows][rows];
        for(int i =0;i<rows;i++){
            triangle[i][0]=1;
            triangle[i][i]=1;
            for(int j=1;j<i;j++){
                triangle[i][j]=triangle[i-1][j-1]+triangle[i-1][j];
            }
        }
        for(int i=0;i<rows;i++){
            for(int j=0;j<=i;j++){
                System.out.print(triangle[i][j]+" ");
            }
            System.out.println();
        }
    }
}
